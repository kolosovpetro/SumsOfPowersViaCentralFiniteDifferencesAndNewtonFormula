\begin{theorem}[Newton series via central difference]
    \begin{align*}
        f(x) &= \sum_{k=0}^{\infty} \frac{\centralFactorial{x}{k}}{k!} \delta^{k} f(0)
    \end{align*}
\end{theorem}

\begin{theorem}[Newton series for power via central difference]
    \begin{align*}
        x^n = \sum_{k=0}^{\infty} \frac{\centralFactorial{x}{k}}{k!} \delta^{k} 0^n
    \end{align*}
\end{theorem}

Where

\begin{corollary}[Central difference of power in zero]
    \label{corollary:central-difference-of-power-in-zero}
    \begin{align*}
        \delta^{k} 0^{n}
        = \sum_{j=0}^{k} (-1)^{j} \binom{k}{j}
        \left(\frac{k}{2} - j\right)^{n}
    \end{align*}
\end{corollary}

\begin{lemma}[Central factorial]
    \begin{align*}
        \centralFactorial{n}{k}
        = n \left( n + \frac{k}{2} -1 \right)\left( n + \frac{k}{2} -2 \right) \cdots \left( n - \frac{k}{2} +1 \right)
        = n \prod_{j=1}^{k-1} \left( n + \frac{k}{2} -j \right)
    \end{align*}
\end{lemma}
Such that
\begin{proposition}
    \begin{align*}
        \centralFactorial{n}{k} = n \fallingFactorial{n+\frac{k}{2}-1}{k-1}
    \end{align*}
\end{proposition}

\begin{proposition}
    For $k \geq 1$
    \begin{align*}
        \frac{\centralFactorial{n}{k}}{k!}
        &= \frac{1}{k!} n \fallingFactorial{n+\frac{k}{2}-1}{k-1}
        = \frac{n}{k (k-1)!} \fallingFactorial{n+\frac{k}{2}-1}{k-1} \\
        &= \frac{n}{k} \binom{n+\frac{k}{2}-1}{k-1}
    \end{align*}
\end{proposition}

Because

\begin{proposition}
    \begin{align*}
        \frac{\fallingFactorial{x}{n}}{n!} = \binom{x}{n}
    \end{align*}
\end{proposition}

Thus
\begin{proposition}
    \begin{align*}
        x^n = \sum_{k=1}^{n} \frac{x}{k} \binom{x+\frac{k}{2}-1}{k-1} \delta^{k} 0^n
    \end{align*}
\end{proposition}
We start from $k=1$ there because for all $k=0$ the central difference $\delta^{k} 0^n$ is zero.

Thus

\begin{proposition}
    \begin{align*}
        x^{n-1} = \sum_{k=1}^{n} \frac{1}{k} \binom{x+\frac{k}{2}-1}{k-1} \delta^{k} 0^n
    \end{align*}
\end{proposition}

Thus

\begin{proposition}
    \begin{align*}
        x^{n} = \sum_{k=0}^{n} \frac{1}{k+1} \binom{x+\frac{k+1}{2}-1}{k} \delta^{k+1} 0^{n+1}
    \end{align*}
\end{proposition}

Thus

\begin{proposition}
    \begin{align*}
        \KnuthRFoldSum{r}{n}{m} = \sum_{k=0}^{m} \frac{1}{k+1} \binom{n+\frac{k+1}{2}-1+r}{k+r} \delta^{k+1} 0^{m+1}
    \end{align*}
    \begin{proof}
        By hockey stick identity.
    \end{proof}
\end{proposition}

\begin{lemma} [Central factorial numbers]
    \begin{align*}
        \delta^k 0^n = k! T(n, k)
    \end{align*}
\end{lemma}

Thus

\begin{proposition}
    \begin{align*}
        \KnuthRFoldSum{r}{n}{m} = \sum_{k=0}^{m} \frac{1}{k+1} \binom{n+\frac{k+1}{2}-1+r}{k+r} (k+1)! T(m+1, k+1)
    \end{align*}
\end{proposition}

Thus

\begin{proposition}
    \begin{align*}
        \KnuthRFoldSum{r}{n}{m-1} = \sum_{k=0}^{m} k! \binom{n+\frac{k+1}{2}-1+r}{k+r} T(m, k+1)
    \end{align*}
\end{proposition}
